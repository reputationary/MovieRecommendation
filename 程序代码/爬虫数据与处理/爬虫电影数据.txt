//创建表
CREATE TABLE movies (
    id INT AUTO_INCREMENT PRIMARY KEY,  # 自增的电影id
    title VARCHAR(255),                 # 电影标题
    src VARCHAR(255),                   # 电影链接
    director VARCHAR(255),              # 导演主演
    score float,                 		# 评分
    commentPeople int,             	  # 评价人数
    summary TEXT,                       # 电影简介
    img_url VARCHAR(255),                # 图片链接
    year int,                           #电影年份
    country VARCHAR(255),     #电影国家
    type VARCHAR(255)                #电影类型
);
